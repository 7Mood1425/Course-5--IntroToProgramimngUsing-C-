#include<iostream>
using namespace std;
#include<fstream>
#include<vector>
#include<string>

void LoadDataFromFileToVector(string FileName, vector<string>& vFileContant) {

	fstream MyFile;
	MyFile.open(FileName, ios::in);
	if (MyFile.is_open())
	{
		string Line;
		while (getline(MyFile, Line))
		{
			vFileContant.push_back(Line);
		}
		MyFile.close();
	}

}

void SaveVectorToFile(string FileName, vector<string>VFile) {

	fstream MyFile;
	MyFile.open(FileName, ios::out);
	if (MyFile.is_open()) 
	{

		for (string &Line : VFile)
		{
			if (Line != "") 
			{
				MyFile << Line << endl;
			}
		}
	}
}
